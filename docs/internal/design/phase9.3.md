下面给 **Phase 9.3：Auto Docs from aiMetadata / Registry MVP** 的详细设计与完整代码。

当前根工程已经有 `docs:dev / docs:build / docs:preview`，但还没有 docs 自动生成脚本。
workspace 已经包含 `apps/*`、`examples/*`，所以 docs/example 仍然可以继续按 workspace 包管理。
Phase 9.3 的核心是：**组件文档不要继续手写 API 表，而是从 `packages/ai/src/metadata.ts` 和 `packages/registry/registry.json` 生成**，避免后续新增组件时文档、AI metadata、registry 三份数据漂移。

# Phase 9.3 目标

```txt
Phase 9.3：Auto Docs from aiMetadata / Registry MVP

目标：
1. 新增 docs 自动生成脚本。
2. 从 @zeus-web/ai metadata 生成 components 文档。
3. 从 registry.json 补充 dependencies/files/sourceTarget 信息。
4. 生成 apps/docs/components/index.md。
5. 生成 apps/docs/components/button.md 等组件页。
6. 增加 docs:generate。
7. 增加 docs:check-generated，校验生成文档是否最新。
8. docs:check 里接入自动文档检查。
9. 不再手动维护组件 API 表。
```

Phase 9.3 不做：

```txt
不生成 guide 文档。
不生成 examples 文档。
不生成 Playground。
不生成 Vue/WC 独立 API 页。
不从源码 AST 解析 props/events。
不改 @zeus-web/ai metadata 结构。
```

# 2. 新增生成器核心

## `scripts/docs/component-docs.ts`

````ts
import type {
  ZeusWebAiComponent,
  ZeusWebAiMetadata,
} from '../../packages/ai/src'
import type { Registry, RegistryItem } from '../../packages/registry/src'

import { readFileSync } from 'node:fs'
import { resolve } from 'node:path'

import { aiMetadata, validateAiMetadata } from '../../packages/ai/src'

export interface ComponentDocsContext {
  metadata: ZeusWebAiMetadata
  registry: Registry
}

export interface GeneratedDoc {
  path: string
  content: string
}

const generatedHeader = `<!-- This file is generated by scripts/commands/generate-docs.ts. Do not edit manually. -->`

export function loadRegistry(root = process.cwd()): Registry {
  return JSON.parse(
    readFileSync(resolve(root, 'packages/registry/registry.json'), 'utf-8'),
  ) as Registry
}

export function createComponentDocsContext(
  root = process.cwd(),
): ComponentDocsContext {
  const result = validateAiMetadata(aiMetadata)

  if (!result.valid) {
    throw new Error(
      [
        'Invalid @zeus-web/ai metadata:',
        ...result.errors.map(error => `- ${error}`),
      ].join('\n'),
    )
  }

  return {
    metadata: aiMetadata,
    registry: loadRegistry(root),
  }
}

function findRegistryItem(
  registry: Registry,
  component: ZeusWebAiComponent,
): RegistryItem | undefined {
  return registry.items.find(item => item.name === component.name)
}

function mdInlineCode(value: string): string {
  return `\`${value.replaceAll('`', '\\`')}\``
}

function mdCodeBlock(language: string, code: string): string {
  return ['```' + language, code, '```'].join('\n')
}

function renderStringList(items: string[]): string {
  if (items.length === 0) {
    return '- None.'
  }

  return items.map(item => `- ${item}`).join('\n')
}

function renderCodeList(items: string[]): string {
  if (items.length === 0) {
    return '- None.'
  }

  return items.map(item => `- ${mdInlineCode(item)}`).join('\n')
}

function renderPropsTable(component: ZeusWebAiComponent): string {
  if (component.props.length === 0) {
    return 'No public props documented.'
  }

  return [
    '| Prop | Type | Default | Description |',
    '| --- | --- | --- | --- |',
    ...component.props.map(prop => {
      const type = mdInlineCode(prop.type)
      const values = prop.values?.length
        ? `<br />Values: ${prop.values.map(mdInlineCode).join(', ')}`
        : ''
      const defaultValue = prop.default ? mdInlineCode(prop.default) : '-'

      return `| ${mdInlineCode(prop.name)} | ${type}${values} | ${defaultValue} | ${prop.description} |`
    }),
  ].join('\n')
}

function renderEventsTable(component: ZeusWebAiComponent): string {
  if (component.events.length === 0) {
    return 'No public events documented.'
  }

  return [
    '| Event | React prop | Detail | Description |',
    '| --- | --- | --- | --- |',
    ...component.events.map(event => {
      const detail = Object.entries(event.detail)
        .map(([key, value]) => `${key}: ${value}`)
        .join(', ')

      return `| ${mdInlineCode(event.name)} | ${mdInlineCode(event.reactName)} | ${mdInlineCode(detail)} | ${event.description} |`
    }),
  ].join('\n')
}

function renderSlotsTable(component: ZeusWebAiComponent): string {
  if (component.slots.length === 0) {
    return 'No slots documented.'
  }

  return [
    '| Slot | Description |',
    '| --- | --- |',
    ...component.slots.map(slot => {
      return `| ${mdInlineCode(slot.name)} | ${slot.description} |`
    }),
  ].join('\n')
}

function renderExamples(component: ZeusWebAiComponent): string {
  if (component.examples.length === 0) {
    return 'No examples documented.'
  }

  return component.examples
    .map(example => {
      return [
        `### ${example.title}`,
        '',
        mdCodeBlock('tsx', example.code),
      ].join('\n')
    })
    .join('\n\n')
}

function renderRegistryInfo(
  component: ZeusWebAiComponent,
  registryItem?: RegistryItem,
): string {
  if (!registryItem) {
    return [
      '## Registry',
      '',
      `Registry item for ${mdInlineCode(component.name)} was not found.`,
    ].join('\n')
  }

  return [
    '## Registry',
    '',
    `Type: ${mdInlineCode(registryItem.type)}`,
    '',
    'Dependencies:',
    '',
    renderCodeList(registryItem.dependencies ?? []),
    '',
    'Files:',
    '',
    registryItem.files.length === 0
      ? '- None.'
      : registryItem.files
          .map(file => {
            return `- ${mdInlineCode(file.path)} → ${mdInlineCode(file.target)} (${mdInlineCode(file.type)})`
          })
          .join('\n'),
  ].join('\n')
}

export function renderComponentDoc(
  component: ZeusWebAiComponent,
  registryItem?: RegistryItem,
): string {
  return [
    generatedHeader,
    '',
    `# ${component.name[0].toUpperCase()}${component.name.slice(1)}`,
    '',
    component.description,
    '',
    '<div class="zw-badge-row">',
    `  <span class="zw-badge">${component.primitivePackage}</span>`,
    `  <span class="zw-badge">${component.registryCommand}</span>`,
    '  <span class="zw-badge">Generated</span>',
    '</div>',
    '',
    '## Add',
    '',
    mdCodeBlock('bash', component.registryCommand),
    '',
    '## Imports',
    '',
    'Styled component:',
    '',
    mdCodeBlock('tsx', component.styledImport),
    '',
    'Primitive React wrapper:',
    '',
    mdCodeBlock('tsx', component.reactImport),
    '',
    'Web Component entry:',
    '',
    mdCodeBlock('ts', component.webComponentImport),
    '',
    '## Props',
    '',
    renderPropsTable(component),
    '',
    '## Events',
    '',
    renderEventsTable(component),
    '',
    '## Slots',
    '',
    renderSlotsTable(component),
    '',
    '## Styling',
    '',
    `Uses Tailwind: ${component.styling.usesTailwind ? 'yes' : 'no'}`,
    '',
    'Theme tokens:',
    '',
    renderCodeList(component.styling.themeTokens),
    '',
    'Internal selectors:',
    '',
    renderCodeList(component.styling.internalSelectors),
    '',
    renderRegistryInfo(component, registryItem),
    '',
    '## AI usage rules',
    '',
    'Do:',
    '',
    renderStringList(component.aiRules.do),
    '',
    'Do not:',
    '',
    renderStringList(component.aiRules.dont),
    '',
    '## Examples',
    '',
    renderExamples(component),
    '',
  ].join('\n')
}

export function renderComponentsIndex(context: ComponentDocsContext): string {
  return [
    generatedHeader,
    '',
    '# Components',
    '',
    'This section is generated from `@zeus-web/ai` metadata and `@zeus-web/registry`.',
    '',
    '| Component | Package | Add command | Description |',
    '| --- | --- | --- | --- |',
    ...context.metadata.components.map(component => {
      return `| [${component.name}](/components/${component.name}) | ${mdInlineCode(component.primitivePackage)} | ${mdInlineCode(component.registryCommand)} | ${component.description} |`
    }),
    '',
    '## Recommended workflow',
    '',
    renderStringList(context.metadata.recommendedWorkflow),
    '',
  ].join('\n')
}

export function generateComponentDocs(
  context = createComponentDocsContext(),
): GeneratedDoc[] {
  const docs: GeneratedDoc[] = [
    {
      path: 'apps/docs/components/index.md',
      content: renderComponentsIndex(context),
    },
  ]

  for (const component of context.metadata.components) {
    docs.push({
      path: `apps/docs/components/${component.name}.md`,
      content: renderComponentDoc(
        component,
        findRegistryItem(context.registry, component),
      ),
    })
  }

  return docs
}
````

# 3. 新增生成命令

## `scripts/commands/generate-docs.ts`

```ts
import { mkdir, writeFile } from 'node:fs/promises'
import { dirname, resolve } from 'node:path'

import pc from 'picocolors'

import { generateComponentDocs } from '../docs/component-docs'

async function main(): Promise<void> {
  const docs = generateComponentDocs()

  for (const doc of docs) {
    const file = resolve(process.cwd(), doc.path)

    await mkdir(dirname(file), { recursive: true })
    await writeFile(file, doc.content, 'utf-8')

    console.log(pc.green(`Generated ${doc.path}`))
  }
}

main().catch(error => {
  console.error(pc.red((error as Error).message))
  process.exit(1)
})
```

# 4. 新增生成文档检查

## `scripts/checks/check-generated-docs.ts`

```ts
import { existsSync, readFileSync } from 'node:fs'
import { resolve } from 'node:path'

import pc from 'picocolors'

import { generateComponentDocs } from '../docs/component-docs'

function main(): void {
  const errors: string[] = []
  const docs = generateComponentDocs()

  for (const doc of docs) {
    const file = resolve(process.cwd(), doc.path)

    if (!existsSync(file)) {
      errors.push(`Missing generated doc: ${doc.path}`)
      continue
    }

    const current = readFileSync(file, 'utf-8')

    if (current !== doc.content) {
      errors.push(`Generated doc is out of date: ${doc.path}`)
    }
  }

  if (errors.length > 0) {
    console.error(pc.red('Generated docs check failed:'))

    for (const error of errors) {
      console.error(`- ${error}`)
    }

    console.error('')
    console.error('Run:')
    console.error('  pnpm docs:generate')
    process.exit(1)
  }

  console.log(pc.green('Generated docs check passed.'))
}

main()
```

# 7. 更新 docs contract check

如果你已有 `scripts/checks/check-docs.ts`，因为组件页现在是生成的，需要确保它们包含 generated header。

在 `requiredDocs` 里给组件页加这个 mustContain：

```ts
'This file is generated by scripts/commands/generate-docs.ts'
````

示例：

```ts
{
  path: 'components/button.md',
  mustContain: [
    'This file is generated by scripts/commands/generate-docs.ts',
    '# Button',
    'zweb add button',
    '@/components/ui/button',
    '@zeus-web/button/react',
  ],
}
```

也建议新增一个专门检查：

```ts
function checkGeneratedComponentDocs(): string[] {
  const components = ['button', 'input', 'checkbox', 'switch', 'tabs', 'dialog']
  const errors: string[] = []

  for (const component of components) {
    const file = resolve(docsRoot, `components/${component}.md`)

    if (!existsSync(file)) {
      errors.push(`Missing generated component doc: components/${component}.md`)
      continue
    }

    const source = readFileSync(file, 'utf-8')

    if (
      !source.includes(
        'This file is generated by scripts/commands/generate-docs.ts',
      )
    ) {
      errors.push(`components/${component}.md must be generated`)
    }
  }

  return errors
}
```

然后在 main 里：

```ts
errors.push(...checkGeneratedComponentDocs())
```
